Implement iterable object via generator

>>> class count(object):
...   def __init__(self,n):
...     self._n = n
...   def __iter__(self):
...     n = self._n
...     while n>0:
...       yield n
...       n-=1
...   def __reversed__(self):
...     n = 0
...     while n<self._n:
...       yield n
...       n+=1
... 
>>> for _ in count(5):
...   print(_, end=" ")
... 
5 4 3 2 1 >>> 
>>> for _ in reversed(count(5)):
...   print(_, end=" ")
... 
0 1 2 3 4 >>>

# delegating list
>>> def chain(a,b):
...   yield from a
...   yield from b
...

Implement a simple scheduler via generator

>>> def fib(n):
...   if n<=2:
...     return 1
...   return fib(n-1)+fib(n-2)
... 
>>> def g_fib(n):
...   for _ in range(1,n+1):
...     yield fib(_)
... 
>>> from collections import deque
>>> t = [g_fib(3),g_fib(5)]
>>> q = deque()
>>> q.extend(t)

What @contextmanager actually doing?

# ref: PyCon 2014 - David Beazley
# define a context manager class
class GeneratorCM(object):
   def __init__(self,gen):
      self._gen = gen
   def __enter__(self):
      return next(self._gen)

   def __exit__(self, *exc_info):
      try:
         if exc_info[0] is None:
            next(self._gen)
         else:
            self._gen.throw(*exc_info)
         raise RuntimeError
      except StopIteration:
         return True
      except:
         raise
            
# define a decorator
def contextmanager(func):
   def run(*a, **k):
      return GeneratorCM(func(*a, **k))
   return run

# example of context manager
@contextmanager
def mylist():
   try:
      l=[1,2,3,4,5]
      yield l
   finally:
      print "exit scope"

with mylist() as l:
   print l

# bash> python example.py 
# [1, 2, 3, 4, 5]

# wrapping generator with "Task"
class Task(object):
   def __init__(self,gen):
      self._gen = gen

   def step(self, val=None):
      # Purpose: run the next yield
      try:
         fut = self._gen.send(val)
         fut.add_done_callback(self._wkup)
      except StopIteration:
         pass
   def _wkup(self, fut):
      # Purpose: handling result
      res = fut.result()
      # "send result" to generator 
      self.step(res)

Do some task after some task done

# ex.py
# wrapping generator with "Task"
class Task(object):
   def __init__(self,gen):
      self._gen = gen

   def step(self, val=None):
      # Purpose: run the next yield
      try:
         fut = self._gen.send(val)
         fut.add_done_callback(self._wakeup)
      except StopIteration:
         pass
   def _wakeup(self, fut):
      # Purpose: handling result
      res = fut.result()
      self.step(res)

# Demo how to use "Task"
import concurrent.futures as cf
import time
pool = cf.ThreadPoolExecutor(8)

def func(x,y):
   return x + y

def do_func(x,y):
   res = yield pool.submit(func,x,y)
   print("Got:", res)

# brute-force solution
def after_bf(delay,gen):
   yield pool.submit(time.sleep, delay)
   result = None
   try:
      while True:
         fut = gen.send(result)
         result = yield fut
   except StopIteration:
      pass
   
# elegant solution
def after_e(delay,gen):
   # do some task like init
   yield pool.submit(time.sleep, delay)
   # do exactly task after some task
   yield from gen

# bash> python3
>>> from ex import Task
>>> from ex import do_func
>>> from ex import after_bf
>>> from ex import after_e
>>> Task(after_bf(5,do_func(2,3))).step()
>>> Got: 5

>>> Task(after_e(5,do_func(2,3))).step()
>>> Got: 5
