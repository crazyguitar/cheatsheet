Calculate Time Cost of a Function

import time
def CalTime(func):
   def wrapper(*args, **kwargs):
      s = time.time()
      func(*args, **kwargs)
      e = time.time()
      print "Cost time: {0}".format(e-s)
   return wrapper
      
@CalTime
def example(n):
   time.sleep(n)

@CalTime
def exLoop():
   for _ in range(10000):
      pass

if __name__ == '__main__':
   example(5)
   exLoop()

Profiling - How Efficiency of a Function

import cProfile, pstats

def Profiling(func):
   def wrapper(*args, **kwargs):
      pr=cProfile.Profile()
      pr.enable()
      func(*args, **kwargs)
      pr.disable()
      ps = pstats.Stats(pr)
      ps.print_stats()
   return wrapper

@Profiling
def example(n):
   import time
   time.sleep(n)

@Profiling
def exLoop():
   for _ in range(10000):
      pass

if __name__ == '__main__':
   example(5)
   exLoop()

Handle INI Format

# ini is not convenient for list or dict
# only "key value" is more suitable 
# see: python document - ConfigParser
>>> from ConfigParser import ConfigParser
>>> def iniLoad(fp):
...   parser = ConfigParser()
...   parser.readfp(fp)
...   sect = parser.sections()
...   dict_ = {}
...   for s in sect:
...     dict_[s] = dict(parser.items(s))
...   return dict_
... 
>>> def iniDump(ini,fp):
...   parser = ConfigParser()
...   for _s in ini:
...     parser.add_section(_s)
...     for _ in ini[_s].items():
...       parser.set(_s,_[0],_[1])
...   parser.write(fp)
...

from sqlalchemy import create_engine
# import table declarative
from sqlalchemy.ext.declarative \
   import declarative_base
# import table object metadata
from sqlalchemy import \
   Column, Integer, String, DateTime
# import generic function
from sqlalchemy import func
# import sessionmaker
from sqlalchemy.orm import sessionmaker

# declare a table
Base = declarative_base()
class exTable(Base):
   __tablename__ = 'extable'
   id = Column(Integer, primary_key=True)
   intcol = Column(Integer)
   strcol = Column(String(8))
   datecol = Column(
      DateTime, 
      default=func.localtimestamp()
   )

class DBsession(object):
   def __init__(self, engine):
      self._engine = engine 
   def __enter__(self):
      smaker = sessionmaker(
         bind=self._engine
      )
      self._session = smaker()
      return self._session
   def __exit__(self,*exc_info):
      if exc_info[0]:
         import traceback
         traceback.print_exception(
            *exc_info
         )
      self._session.close()

if __name__ == "__main__":
   db = 'sqlite:///example.db'
   # create engine 
   engine = create_engine(db, echo=True)
   # create table
   Base.metadata.create_all(engine)
   with DBsession(engine) as session:
      ex = exTable(intcol=1, strcol="ex")
      session.add(ex)
      session.commit()

A Basic Synchronized Socket Server

import socket

class Socket(object):
   def __init__(self,host,port):
      self._host = host
      self._port = port
   def __enter__(self):
      sock = socket.socket(
         socket.AF_INET,
         socket.SOCK_STREAM)
      sock.setsockopt(
         socket.SOL_SOCKET,
         socket.SO_REUSEADDR,1)
      sock.bind((self._host,self._port))
      sock.listen(5)
      self._sock = sock
      return self._sock
   def __exit__(self,*exc_info):
      if exc_info[0]:
         import traceback
         traceback.print_exception(
            *exc_info)
         self._sock.close()

Select Based Async Socket Server

import select

readList   = [], writeList  = []
exceptList = [], clients    = {}
host = 'localhost'
port = 5566
with server(host, port) as sock:
   readList.append(sock)
   while True:
      """
      ready = (ReadReady,
               WriteReady,
               ExceptionReady)
      """
      ready = select.select(
               readList,
               writeList,
               exceptList)
      # process read ready
      for r in ready[0]:
         if r == sock:
            client, addr = sock.accept()
         else:
            msg = r.recv(4096)
            writeList.append(r)
            clients{r} = msg
      # process write ready
      for w in ready[1]:
         w.send(clients[w])
         writeList.remove(w)
         del clients[w]

Poll Based Async Socket Server

import select

host     = 'localhost'
port     = 5566
READ     = select.POLLIN | select.POLLPRI 
WRITE    = select.POLLOUT
poller   = select.poll()
fd_info  = {sock.fileno():sock}
recv_msg = {} 
poller.register(sock, READONLY)
with server(host, port) as sock:
  while True:
    events = poller.poll()   
    for fd, flag in events:
      s = fd_info[fd]   
      # process read
      if flag & (select.POLLIN|\
                 select.POLLPRI):
        if s == sock:
          conn, addr = sock.accept()
          poller.register(conn, READ)
          fd_info[conn.fileno()] = conn
        else:
          msg = s.recv(4096)
          # check client conect
          if msg:
            recv_msg[s.fileno()] = msg  
            poller.modify(s, WRITE)
          else:
            poller.unregister(s)
            print "client leave"
            s.close()

      if flag & select.POLLOUT:
        s.send(recv_msg[s.fileno()])
        poller.modify(s, READ)
